```java
public class A_反幻方 {
    static int count = 0;
    static int[] P = new int[10000];

    public static void main(String[] args) {
        int A[] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
        digui(A.length, A, 0);
        System.out.println(count / 8);  //TODO 为什么count 不加static 不可以访问, 因为 main 方法是 static
    }

    public static void countMagicSquare() {
        int x1 = P[0] + P[1] + P[2];
        int x2 = P[3] + P[4] + P[5];
        int x3 = P[6] + P[7] + P[8];

        int x4 = P[0] + P[3] + P[6];
        int x5 = P[1] + P[4] + P[7];
        int x6 = P[2] + P[5] + P[8];

        int x7 = P[0] + P[4] + P[8];
        int x8 = P[2] + P[4] + P[6];

        if (x1 != x2 && x1 != x3 && x1 != x4 && x1 != x5 && x1 != x6 && x1 != x7 && x1 != x8)
            if (x2 != x3 && x2 != x4 && x2 != x5 && x2 != x6 && x2 != x7 && x2 != x8)
                if (x3 != x4 && x3 != x5 && x3 != x6 && x3 != x7 && x3 != x8)
                    if (x4 != x5 && x4 != x6 && x4 != x7 && x4 != x8)
                        if (x5 != x6 && x5 != x7 && x5 != x8)
                            if (x6 != x7 && x6 != x8)
                                if (x7 != x8)
                                    count++;
    }

    /**  通用模板, 可用来处理可重集的全排列
     * @param n   要排列的数组的长度
     * @param A   要排列的数组
     * @param cur 当前处理的数组元素的下标
     */
    public static void digui(int n, int[] A, int cur) {
        if (cur == n) {
            countMagicSquare();
        }
        for (int i = 0; i < n; i++)
            if ((i == 0) || A[i] != A[i - 1]) {
                int count1 = 0;
                int count2 = 0;
                for (int j = 0; j < n; j++) if (A[j] == A[i]) count1++;
                for (int k = 0; k < cur; k++) if (P[k] == A[i]) count2++;
                if (count2 < count1) {
                    P[cur] = A[i];
                    digui(n, A, cur + 1);
                }
            }
    }
}
```

// 可重集的全排列
// Rujia Liu
#include<cstdio>
#include<algorithm>
using namespace std;

int P[100], A[100];

// 输出数组P中元素的全排列。数组P中可能有重复元素
void print_permutation(int n, int* P, int* A, int cur) {
  if(cur == n) {
    for(int i = 0; i < n; i++) printf("%d ", A[i]);
    printf("\n");
  } else for(int i = 0; i < n; i++) if(!i || P[i] != P[i-1]) {
    int c1 = 0, c2 = 0;
    for(int j = 0; j < cur; j++) if(A[j] == P[i]) c1++;
    for(int j = 0; j < n; j++) if(P[i] == P[j]) c2++;
    if(c1 < c2) {
      A[cur] = P[i];
      print_permutation(n, P, A, cur+1);
    }
  }
}

int main() {
  int i, n;
  scanf("%d", &n);
  for(i = 0; i < n; i++)
    scanf("%d", &P[i]);
  sort(P, P+n);
  print_permutation(n, P, A, 0);
  return 0;
}

public class A_反幻方 {
    static int count = 0;
    static int[] P = new int[10000];

    public static void main(String[] args) {
        int A[] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
        digui(A.length, A, 0);
        System.out.println(count / 8);  //TODO 为什么count 不加static 不可以访问, 因为 main 方法是 static
    }

    public static void countMagicSquare() {
        int x1 = P[0] + P[1] + P[2];
        int x2 = P[3] + P[4] + P[5];
        int x3 = P[6] + P[7] + P[8];

        int x4 = P[0] + P[3] + P[6];
        int x5 = P[1] + P[4] + P[7];
        int x6 = P[2] + P[5] + P[8];

        int x7 = P[0] + P[4] + P[8];
        int x8 = P[2] + P[4] + P[6];

        if (x1 != x2 && x1 != x3 && x1 != x4 && x1 != x5 && x1 != x6 && x1 != x7 && x1 != x8)
            if (x2 != x3 && x2 != x4 && x2 != x5 && x2 != x6 && x2 != x7 && x2 != x8)
                if (x3 != x4 && x3 != x5 && x3 != x6 && x3 != x7 && x3 != x8)
                    if (x4 != x5 && x4 != x6 && x4 != x7 && x4 != x8)
                        if (x5 != x6 && x5 != x7 && x5 != x8)
                            if (x6 != x7 && x6 != x8)
                                if (x7 != x8)
                                    count++;
    }

    /**  通用模板, 可用来处理可重集的全排列
     * @param n   要排列的数组的长度
     * @param A   要排列的数组
     * @param cur 当前处理的数组元素的下标
     */
    public static void digui(int n, int[] A, int cur) {
        if (cur == n) {
            countMagicSquare();
            return;  // 不要忘记
        }
        for (int i = 0; i < n; i++)
            if ((i == 0) || A[i] != A[i - 1]) {
                int count1 = 0;
                int count2 = 0;
                for (int j = 0; j < n; j++) if (A[j] == A[i]) count1++;
                for (int k = 0; k < cur; k++) if (P[k] == A[i]) count2++;
                if (count2 < count1) {
                    P[cur] = A[i];
                    digui(n, A, cur + 1);
                }
            }
    }
}
